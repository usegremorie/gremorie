## Usage

<Tabs items={["Angular", "React"]}>

<Tab value="Angular">

Em breve.

</Tab>

<Tab value="React">

```tsx
import {
  Image,
} from '@gremorie/rx-ai';

export function Demo() {
  return (
    <Image
      base64={image.base64}
      uint8Array={image.uint8Array}
      mediaType={image.mediaType}
      alt="Generated chart"
    />
  );
}
```
