import {
  CodeBlock,
  CodeBlockCopyButton,
} from '@gremorie/rx-ai';

  return (

## Notes

- This primitive was migrated from Bridge into `@gremorie/rx-ai` as part of the Fase 5j code batch (Bridge parity for the AI category).
- All cross-Gremorie imports use the published `@gremorie/rx-*` packages; the registry mirrors that into per-primitive source so consumers can `npx gremorie add rx-code-block` without pulling unrelated AI components.
